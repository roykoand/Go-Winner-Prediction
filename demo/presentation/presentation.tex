\section{Model}
\begin{frame}
\frametitle{Optimization parameters and loss function}

\begin{center}
Loss function: Cross entropy

Optimizer: Adam (learning rate = $3 \cdot 10^{-4}$)

Learning rate scheduler: ReduceLROnPlateau
\end{center}

\end{frame}
